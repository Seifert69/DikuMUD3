case DILF_FGT:
            switch (dil_getval(v1))
            {
                case DILV_NULL: /* not applicable */
                case DILV_FAIL:
                    v->type = DILV_FAIL; /* not applicable */
                    break;
                case DILV_UP:
                    if (v1->val.ptr && ((unit_data *)v1->val.ptr)->isChar())
                    {
                        v->atyp = DILA_NONE;
                        v->type = DILV_UP;
                        v->val.ptr = CHAR_FIGHTING((unit_data *)v1->val.ptr);
                    }
                    else
                    {
                        v->type = DILV_FAIL;
                    }
                    break;
                default:
                    v->type = DILV_ERR; /* wrong type */
                    break;
            }
            break;
