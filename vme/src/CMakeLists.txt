target_include_directories(vme_objs
        PRIVATE .
        PUBLIC ${CMAKE_SOURCE_DIR}/vme/include)

target_compile_definitions(vme PUBLIC DMSERVER LINUX POSIX)
target_include_directories(vme_objs PUBLIC ${CMAKE_SOURCE_DIR}/vme/include)

        vme_objs
        crypt
        Boost::filesystem
        Boost::regex
        pthread
