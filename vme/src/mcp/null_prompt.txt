void dilfe_null(dilprg *p)
{
    dilval *v = new dilval;
    /* Pointer value null */

    v->type = DILV_NULL;
    v->atyp = DILA_NONE;
    v->val.ptr = nullptr;
    p->stack.push(v);
}
