Generate MCP entry for DIL keyword: npcs

=== Yacc rule ===
| DILSF_NPCS /* .npcs */
{
    INITEXP($$);
    $$.rtyp = DilVarType_e::DILV_ZP;
    $$.typ = DilVarType_e::DILV_UP;
    $$.dsl = DSL_DYN;
    $$.num = DILF_NPCS;
}

=== C implementation ===
case DILF_NPCS:
            switch (dil_getval(v1))
            {
                case DILV_NULL: /* not applicable */
                case DILV_FAIL:
                    v->type = DILV_FAIL; /* not applicable */
                    break;
                case DILV_ZP:
                    if (v1->val.ptr)
                    {
                        v->atyp = DILA_NORM;
                        v->type = DILV_UP;
                        auto *z = (zone_type *)v1->val.ptr;

                        v->val.ptr = z->findFirstNPC();
                        if (v->val.ptr == nullptr)
                        {
                            v->type = DILV_FAIL;
                        }
                    }
                    else
                    {
                        v->type = DILV_FAIL; /* not applicable */
                    }
                    break;
                default:
                    v->type = DILV_ERR; /* wrong type */
                    break;
            }
            break;
