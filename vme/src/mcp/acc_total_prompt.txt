=== Yacc rule ===
| DILSF_ATOT /* .acc_total */
{
    INITEXP($$);
    $$.rtyp = DilVarType_e::DILV_UP;
    $$.typ = DilVarType_e::DILV_INT;
    $$.dsl = DSL_DYN;
    $$.num = DILF_ATOT;
}

=== C implementation ===
case DILF_ATOT:
            switch (dil_getval(v1))
            {
                case DILV_NULL:
                case DILV_FAIL:
                    v->type = DILV_FAIL; /* not applicable */
                    break;
                case DILV_UP:
                {
                    auto *player = reinterpret_cast<pc_data *>(v1->val.ptr);
                    if (player && player->isPC())
                    {
                        v->atyp = DILA_NORM;
                        v->type = DILV_INT;
                        if (g_cServerConfig.isAccounting())
                        {
                            v->val.num = player->getPCAccountData().getTotalCredit();
                        }
                        else
                        {
                            v->val.num = 0;
                        }
                    }
                    else
                    {
                        v->type = DILV_FAIL; /* not applicable */
                    }
                }
                break;
                default:
                    v->type = DILV_ERR; /* wrong type */
                    break;
            }
            break;
